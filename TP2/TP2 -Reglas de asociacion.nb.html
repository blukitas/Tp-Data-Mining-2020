tweets <- mongo(collection = "tweets_tp2", db = "DMUBA")

nrow(df_tweets)

#preprocesamiento #se calcula por cada tweet retweeteado, #el tiempo entre la creacion del tweet y su primer RT >tiempo_primera_rpta_RT #el tiempo entre el primer RT y el ultimo RT > duracion_RT #el tiempo entre la creacion del tweet y el ultimo RT > vida_RT #cantidad de RT > cant_RT

stringsqla = "select retweet_status_id, retweet_created_at as fecha_inicio_T, min(created_at) as fecha_primer_RT, max(created_at) as fecha_ultimo_RT,min(created_at) - retweet_created_at as tiempo_primera_rpta_RT,max(created_at) - min(created_at) as duracion_RT, max(created_at) - retweet_created_at as vida_RT, count(*) as cant_RT from df_tweets group by retweet_status_id"

df_tweets_tot = sqldf(stringsqla)
